KXWD_BEL_contest_active_GER = {
	tag = GER
	is_ai = yes
	has_global_flag = KXWD_belgian_influence_active

}

	KXWD_BEL_contest_active_GER = yes
	check_variable = { KXWD_BEL_active_prep_missions < 3 }

}
