defined_text = {
	name = GetHOLBMProgressStatus
	text = {
		trigger = { check_variable = { var = hol_bm_progress value = 25 compare = less_than } }
		localization_key = HOL_bm_status_failing
	}
	text = {
		trigger = { check_variable = { var = hol_bm_progress value = 80 compare = greater_than_or_equals } }
		localization_key = HOL_bm_status_thriving
	}
	text = {
		localization_key = HOL_bm_status_normal
	}
}
